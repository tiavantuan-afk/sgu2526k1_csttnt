Hin th ngu nhin $N=50$ hnh trn:

  • Tm $c_i = (x_i,y_i)$ ngu nhin vi $x_i, y_i \in \left[0, 1 \right]$
  • Mu sc $color_i$ ngu nhin vi $color_i \in \left[0, 1 \right]$
  • Din tch $area_i = \pi * r_i^2$ vi sinh ngu nhin $r_i \in \left[0, 1 \right]$

N = 50 # S lng hnh trn

# Sinh ta  tm (x, y) trong khong [0, 1]
x = np.random.rand(N)
y = np.random.rand(N)

# Sinh mu sc ngu nhin (gi tr t 0 n 1)
colors = np.random.rand(N)

# Sinh din tch ngu nhin
# Bn knh r t 0 n 15 (nhn 15  hnh to r hn trn biu )
r = 15 * np.random.rand(N)
area = np.pi * (r ** 2)

# 3. Kim tra d liu 
print(f' sinh {N} hnh trn:')
print(f'- Ta  X (min-max): {min(x):.2f} - {max(x):.2f}')
print(f'- Ta  Y (min-max): {min(y):.2f} - {max(y):.2f}')
print(f'- Din tch ln nht: {max(area):.2f}')

# 4. Hin th biu  (Scatter Plot)
plt.figure(figsize=(8, 6)) # Thit lp kch thc khung hnh
plt.scatter(x, y, s=area, c=colors, alpha=0.5, cmap='viridis')

# Trang tr biu 
plt.title('Biu  50 hnh trn ngu nhin')
plt.xlabel('Trc X')
plt.ylabel('Trc Y')
plt.grid(True, linestyle='--', alpha=0.3) # Thm li m
plt.colorbar(label='Thang mu') # Hin thanh mu bn cnh

# Hin th
plt.show()
